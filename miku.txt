  cd /mnt/d/Github/deer-flow
  make docker-stop

  # 方式二：直接 docker compose 停止
  cd /mnt/d/Github/deer-flow
  docker compose -f docker/docker-compose-dev.yaml down

  # 方式三：停止并删除容器+网络（完全清理）
  cd /mnt/d/Github/deer-flow
  docker compose -f docker/docker-compose-dev.yaml down --volumes --remove-orphans

  2. 查看当前状态

  # 查看运行中的容器
  docker ps

  # 查看所有容器（包括已停止的）
  docker ps -a

  # 查看 DeerFlow 相关容器
  docker ps --filter "name=deer-flow"

  3. 清理 Docker 资源

  # 删除已停止的容器
  docker container prune -f

  # 删除无用镜像（释放磁盘空间）
  docker image prune -a

  # 完全清理：删除停止的容器、无用镜像、无用网络
  docker system prune -a --volumes

  4. 重新启动 DeerFlow

  cd /mnt/d/Github/deer-flow

  # 重新启动（保留数据）
  docker compose -f docker/docker-compose-dev.yaml up -d

  # 完全重启（清理旧容器和网络）
  docker compose -f docker/docker-compose-dev.yaml down
  docker compose -f docker/docker-compose-dev.yaml up -d

  5. 查看日志

  # 查看所有容器日志

  # 实时查看 gateway 日志

  # 查看最近 100 行日志
  docker logs --tail 100 deer-flow-gateway
